Product Overview

Vision Statement

Zenith AI is the operational revenue infrastructure layer that dental practices install once and run forever. We exist at the intersection of AI-powered systems and dental operations — not as a tool, but as infrastructure that silently recovers revenue, fills chairs, and frees practitioners to focus on clinical excellence.

Core Thesis: The average dental practice loses $8,000–$15,000/month to preventable operational failures — no-shows, lapsed recall patients, unreviewed care, and front-desk overload. Zenith AI installs the Patient Revenue Engine™ to systematically recover that revenue without adding headcount.

Problem Being Solved

Revenue Leakage

  • No-show rates averaging 18–25% across solo/group practices
  • Recall gap: 40–60% of patients due for recall never return
  • Each empty chair = $350–$800 lost per appointment slot
  • $120K–$180K in annual preventable revenue loss

Operational Overload

  • Front desk spending 4–6 hours/day on manual outreach
  • No systematic follow-up for 80% of lapsed patients
  • Review generation manual, inconsistent, and often forgotten
  • Intake paperwork still largely paper-based or siloed

Patient Revenue Engine™

The Patient Revenue Engine™ is the core operational system — a multi-stage, automated patient lifecycle management system that operates continuously across four primary revenue recovery channels.

Revenue Recovery Channels

01
No-Show Prevention

Triggered 48hr, 24hr, 2hr before appointment. SMS + email. Confirmation required or slot opened.

02
Recall Recovery

Patients 90, 180, 365 days overdue trigger outreach sequence. 3-touch cadence per segment.

03
Review Generation

Post-visit (4hr delay) SMS requesting review. Negative sentiment filtered before public routing.

04
Intake Optimization

Digital pre-visit forms reduce chair time waste. Data auto-populates into PMS where supported.

Lifecycle State Machine

PATIENT STATES
──────────────────────────────────
new_patient
  → intake_sent
  → intake_complete
  → appointment_scheduled
    → reminder_48hr_sent
    → reminder_24hr_sent
    → reminder_2hr_sent
    → confirmed | unresponsive
  → attended → review_requested
  → no_show → recall_queue_7d

recall_queue (3mo gap)
  → outreach_1_sent
  → outreach_2_sent (14d)
  → outreach_3_sent (30d)
  → rebooked | lapsed_final

CREATE TYPE patient_status AS ENUM (
  'active', 'lapsed_90', 'lapsed_180', 'lapsed_365', 'inactive', 'archived'

  'reminder_48hr', 'reminder_24hr', 'reminder_2hr',
  'recall_90', 'recall_180', 'recall_365',
  'review_request', 'intake_form', 'reactivation'

-- PRACTICES (Multi-tenant root)
CREATE TABLE practices (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name           TEXT NOT NULL,
  slug           TEXT UNIQUE NOT NULL,
  owner_email    TEXT NOT NULL,
  phone          TEXT,
  address        TEXT,
  city           TEXT,
  state          TEXT,
  zip            TEXT,
  pms_software   TEXT, -- 'dentrix' | 'eaglesoft' | 'opendental' | 'other'
  subscription   TEXT NOT NULL DEFAULT 'trial',
  trial_ends_at  TIMESTAMPTZ,
  settings       JSONB DEFAULT '{}',
  is_active      BOOLEAN DEFAULT true,
  created_at     TIMESTAMPTZ DEFAULT now(),
  updated_at     TIMESTAMPTZ DEFAULT now()

-- PATIENTS
CREATE TABLE patients (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  practice_id     UUID NOT NULL REFERENCES practices(id) ON DELETE CASCADE,
  first_name      TEXT NOT NULL,
  last_name       TEXT NOT NULL,
  email           TEXT,
  phone           TEXT,
  dob             DATE,
  last_visit      DATE,
  next_recall_due DATE,
  status          patient_status DEFAULT 'active',
  pms_id          TEXT, -- external PMS identifier
  metadata        JSONB DEFAULT '{}',
  created_at      TIMESTAMPTZ DEFAULT now(),
  updated_at      TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_patients_practice ON patients(practice_id);
CREATE INDEX idx_patients_status ON patients(status);
CREATE INDEX idx_patients_recall ON patients(next_recall_due);

-- APPOINTMENTS
CREATE TABLE appointments (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  practice_id     UUID NOT NULL REFERENCES practices(id) ON DELETE CASCADE,
  patient_id      UUID NOT NULL REFERENCES patients(id) ON DELETE CASCADE,
  scheduled_at    TIMESTAMPTZ NOT NULL,
  duration_mins   INTEGER NOT NULL DEFAULT 60,
  procedure_type  TEXT,
  provider_name   TEXT,
  chair_number    INTEGER,
  status          appointment_status DEFAULT 'scheduled',
  confirmed_at    TIMESTAMPTZ,
  no_show_at      TIMESTAMPTZ,
  revenue_value   NUMERIC(10,2),
  notes           TEXT,
  created_at      TIMESTAMPTZ DEFAULT now(),
  updated_at      TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_appt_practice ON appointments(practice_id);
CREATE INDEX idx_appt_patient ON appointments(patient_id);
CREATE INDEX idx_appt_scheduled ON appointments(scheduled_at);
CREATE INDEX idx_appt_status ON appointments(status);

-- OUTREACH LOG
CREATE TABLE outreach_events (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  practice_id     UUID NOT NULL REFERENCES practices(id) ON DELETE CASCADE,
  patient_id      UUID NOT NULL REFERENCES patients(id) ON DELETE CASCADE,
  appointment_id  UUID REFERENCES appointments(id),
  outreach_type   outreach_type NOT NULL,
  channel         outreach_channel NOT NULL,
  sent_at         TIMESTAMPTZ DEFAULT now(),
  delivered       BOOLEAN DEFAULT false,
  responded       BOOLEAN DEFAULT false,
  response_at     TIMESTAMPTZ,
  provider_msg_id TEXT,
  payload         JSONB,
  created_at      TIMESTAMPTZ DEFAULT now()
);

-- SYSTEM EVENTS (automation engine)
CREATE TABLE system_events (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  practice_id  UUID REFERENCES practices(id),
  event_type   TEXT NOT NULL,
  payload      JSONB NOT NULL DEFAULT '{}',
  processed    BOOLEAN DEFAULT false,
  created_at   TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_events_type ON system_events(event_type);
CREATE INDEX idx_events_processed ON system_events(processed) WHERE NOT processed;

-- REVIEWS
CREATE TABLE reviews (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  practice_id    UUID NOT NULL REFERENCES practices(id),

  rating         INTEGER CHECK (rating BETWEEN 1 AND 5),
  platform       TEXT, -- 'google' | 'healthgrades' | 'yelp'
  published      BOOLEAN DEFAULT false,
  created_at     TIMESTAMPTZ DEFAULT now()
);

-- LEADS (from website ROI calculator)
CREATE TABLE leads (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  avg_value       NUMERIC(10,2),

  roi_estimate    NUMERIC(10,2),

ROI Calculator Logic

Revenue Leakage Formula

// lib/roi-calculator.ts

export interface PracticeInputs {
  chairs: number;
  monthly_appointments: number;
  avg_appointment_value: number;
  no_show_pct: number;
  recall_lapsed_pct: number;
  admin_hours_per_day: number;
}

export interface ROIOutputs {
  monthly_no_show_loss: number;
  monthly_recall_loss: number;
  monthly_admin_cost: number;
  total_monthly_leakage: number;
  annual_leakage: number;
  recoverable_monthly: number;
  recoverable_annual: number;
  admin_hours_saved: number;
  review_growth_monthly: number;
}

export function calculateROI(inputs: PracticeInputs): ROIOutputs {
  // No-show revenue loss
  const no_show_appts = inputs.monthly_appointments * (inputs.no_show_pct / 100);
  const monthly_no_show_loss = no_show_appts * inputs.avg_appointment_value;

  // Recall revenue loss (lapsed patients who never returned)
  const recall_patients = inputs.monthly_appointments * 0.15;
  const monthly_recall_loss = recall_patients * (inputs.recall_lapsed_pct / 100)
    * inputs.avg_appointment_value;

  // Admin efficiency cost (at $22/hr blended front desk rate)
  const admin_cost_per_hour = 22;
  const monthly_admin_cost = inputs.admin_hours_per_day * 22 * admin_cost_per_hour;

  const total_monthly_leakage = monthly_no_show_loss + monthly_recall_loss + monthly_admin_cost;

  // Recovery rate: 40% no-show, 25% recall, 60% admin efficiency
  const recoverable_monthly =
    (monthly_no_show_loss * 0.40) +
    (monthly_recall_loss * 0.25) +
    (monthly_admin_cost * 0.60);

  return {
    monthly_no_show_loss,
    monthly_recall_loss,
    monthly_admin_cost,
    total_monthly_leakage,
    annual_leakage: total_monthly_leakage * 12,
    recoverable_monthly,
    recoverable_annual: recoverable_monthly * 12,
    admin_hours_saved: inputs.admin_hours_per_day * 0.6 * 22,
    review_growth_monthly: Math.round(inputs.monthly_appointments * 0.05),
  };
}
